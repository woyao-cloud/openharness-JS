**Implementation:**

```ts
register("hooks", "List loaded hooks by event", () => {
  const { getHooks } = require("../harness/hooks.js");
  const hooks = getHooks() ?? {};
  const lines: string[] = ["─── Loaded Hooks ───", ""];
  const events = Object.keys(hooks).sort();
  if (events.length === 0) {
    lines.push("  No hooks configured.");
    lines.push("  Add hooks to .oh/config.yaml under `hooks:`");
    return { output: lines.join("\n") };
  }
  for (const event of events) {
    const defs = hooks[event] as HookDef[];
    lines.push(`  ${event} (${defs.length}):`);
    for (const def of defs) {
      const kind = def.command ? `command` : def.url ? `http` : def.prompt ? `prompt` : "?";
      const match = def.match ? ` [match: ${def.match}]` : "";
      const source = def.command ?? def.url ?? def.prompt ?? "";
      lines.push(`    - ${kind}: ${source.slice(0, 60)}${source.length > 60 ? "…" : ""}${match}`);
    }
  }
  return { output: lines.join("\n") };
});
```

```ts
// src/main.tsx, inside the opts.print branch, after the event loop (line ~729)
if (opts.jsonSchema) {
  let schema: unknown;
  try {
    schema = typeof opts.jsonSchema === "string"
      ? JSON.parse(opts.jsonSchema)
      : opts.jsonSchema;
  } catch (e) {
    process.stderr.write(`[error] --json-schema is not valid JSON: ${(e as Error).message}\n`);
    process.exit(2);
  }

  let parsed: unknown;
  try {
    parsed = JSON.parse(fullOutput.trim());
  } catch (e) {
    process.stderr.write(`[error] Model output is not valid JSON: ${(e as Error).message}\n`);
    process.exit(3);
  }

  // Validate with Zod (convert JSON Schema → Zod via existing helper, or use a minimal JSON-Schema validator)
  const result = validateAgainstJsonSchema(parsed, schema);
  if (!result.ok) {
    process.stderr.write(`[error] Output does not match schema:\n${result.errors.join("\n")}\n`);
    process.exit(4);
  }

  // Emit the validated JSON on stdout as the sole output
  console.log(JSON.stringify(parsed));
  process.exit(0);
}
```

**Implementation note:** If Zod can't ingest JSON Schema directly, use `ajv` (lightweight, already may be transitive) or write a minimal validator covering the common subset (`type`, `properties`, `required`, `items`, `enum`). Keep scope tight — don't build a full JSON Schema engine.

## Risks / things to watch

- **Task 1 precedence rule** — project styles override user styles. Test this explicitly; drift here breaks silently.
- **Task 2 partial-message stream volume** — every token chunk becomes an NDJSON line. At 100 tok/sec that's 100 JSON serializations/second. Acceptable but worth a perf note.
- **Task 5 subagent visibility** — MCP resource tools should inherit from the main agent's MCP connections; subagents spun via `Agent` tool may not see the same servers. Check `src/tools/AgentTool/index.ts` to see how it currently handles MCP propagation.
- **Task 6 JSON Schema library choice** — adding `ajv` is the pragmatic choice but adds a dep. If user prefers zero-dep, write a minimal validator; note in the PR description which you picked.
